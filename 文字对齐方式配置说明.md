# 文字对齐方式配置说明

## 新增属性

### `textAlignment`

控制文字的对齐方式，默认为居中。

```swift
var textAlignment: NSTextAlignment = .center
```

**支持的值：**
- `.left` - 左对齐
- `.center` - 居中对齐（默认）
- `.right` - 右对齐
- `.justified` - 两端对齐
- `.natural` - 自然对齐（根据语言方向）

## 使用场景

### 场景 1：默认居中（无需设置）

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
label.text = "VIP1"
label.enableMarquee = true

// 默认居中，无需设置
// 结果：文字在容器中居中显示
```

### 场景 2：左对齐

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
label.text = "VIP1"
label.textAlignment = .left  // ✅ 设置左对齐
label.enableMarquee = true

// 结果：文字靠左显示
```

### 场景 3：右对齐

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
label.text = "VIP1"
label.textAlignment = .right  // ✅ 设置右对齐
label.enableMarquee = true

// 结果：文字靠右显示
```

### 场景 4：动态修改对齐方式

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
label.text = "VIP1"
label.textAlignment = .center
label.enableMarquee = true

// 稍后修改对齐方式
label.textAlignment = .left  // 立即生效
```

## 在 GMGradientNameView 中使用

### 示例 1：默认居中

```swift
class GMGradientNameView: GMWaveGradientLabelView {
    
    func configUI(font: UIFont, level: Int, defaultColors: [UIColor]) {
        stopAnimation()
        enableMarquee = false
        
        self.gradientDirection = .topRightToBottomLeft
        self.font = font
        self.gradientColors = getColors(level: level, defaultColors: defaultColors)
        
        // ✅ 默认居中，无需设置
        // self.textAlignment = .center
        
        if level > 0 {
            self.animationDuration = 2.0
            self.marqueeSpeed = 20
            self.marqueeDelay = 1.0
            self.marqueeGap = 10
            
            DispatchQueue.main.async { [weak self] in
                self?.enableMarquee = true
            }
        }
    }
}
```

### 示例 2：自定义左对齐

```swift
class GMGradientNameView: GMWaveGradientLabelView {
    
    func configUI(font: UIFont, level: Int, defaultColors: [UIColor], alignment: NSTextAlignment = .center) {
        stopAnimation()
        enableMarquee = false
        
        self.gradientDirection = .topRightToBottomLeft
        self.font = font
        self.gradientColors = getColors(level: level, defaultColors: defaultColors)
        
        // ✅ 支持自定义对齐方式
        self.textAlignment = alignment
        
        if level > 0 {
            self.animationDuration = 2.0
            self.marqueeSpeed = 20
            self.marqueeDelay = 1.0
            self.marqueeGap = 10
            
            DispatchQueue.main.async { [weak self] in
                self?.enableMarquee = true
            }
        }
    }
}

// 使用时
nameLabel.configUI(font: font, level: 1, defaultColors: [.white], alignment: .left)
```

## 不同场景下的效果

### 场景 1：短文字 + 大容器（不滚动）

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
label.text = "VIP1"
label.enableMarquee = true

// 居中（默认）
label.textAlignment = .center
// 结果：[        VIP1        ]

// 左对齐
label.textAlignment = .left
// 结果：[VIP1                ]

// 右对齐
label.textAlignment = .right
// 结果：[                VIP1]
```

### 场景 2：长文字 + 小容器（滚动）

```swift
let label = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 150, height: 60))
label.text = "这是一段很长的文字"
label.enableMarquee = true

// 对齐方式对滚动文字无影响
// 滚动时文字会从右向左移动，完整显示所有内容
```

## 注意事项

### 1. 对齐方式只影响静态显示

当文字需要滚动时，对齐方式不影响滚动效果：
- 滚动时：文字从右向左移动，完整显示
- 不滚动时：根据 `textAlignment` 对齐

### 2. 跑马灯模式下的对齐

```swift
// 启用跑马灯
label.enableMarquee = true

// 如果文字短，不需要滚动
// - textAlignment = .center → 居中显示
// - textAlignment = .left → 左对齐显示
// - textAlignment = .right → 右对齐显示

// 如果文字长，需要滚动
// - 无论 textAlignment 是什么，都会滚动显示完整文字
```

### 3. 动态修改

```swift
// 可以随时修改对齐方式
label.textAlignment = .left
// 立即生效，无需重新设置文字或重启动画
```

## 完整示例

### 示例 1：三种对齐方式对比

```swift
class ViewController: UIViewController {
    
    override func viewDidLoad() {
        super.viewDidLoad()
        view.backgroundColor = .black
        
        // 居中对齐
        let centerLabel = WaveGradientLabelView(frame: CGRect(x: 20, y: 100, width: 300, height: 60))
        centerLabel.text = "居中对齐"
        centerLabel.textAlignment = .center
        centerLabel.gradientColors = [.systemPink, .systemPurple, .systemBlue]
        centerLabel.enableMarquee = true
        view.addSubview(centerLabel)
        
        // 左对齐
        let leftLabel = WaveGradientLabelView(frame: CGRect(x: 20, y: 200, width: 300, height: 60))
        leftLabel.text = "左对齐"
        leftLabel.textAlignment = .left
        leftLabel.gradientColors = [.systemRed, .systemOrange, .systemYellow]
        leftLabel.enableMarquee = true
        view.addSubview(leftLabel)
        
        // 右对齐
        let rightLabel = WaveGradientLabelView(frame: CGRect(x: 20, y: 300, width: 300, height: 60))
        rightLabel.text = "右对齐"
        rightLabel.textAlignment = .right
        rightLabel.gradientColors = [.systemGreen, .systemTeal, .systemCyan]
        rightLabel.enableMarquee = true
        view.addSubview(rightLabel)
    }
}
```

### 示例 2：在 Cell 中使用

```swift
class VIPCell: UITableViewCell {
    let nameLabel = GMGradientNameView()
    
    func configure(name: String, level: Int, alignment: NSTextAlignment = .center) {
        nameLabel.text = name
        nameLabel.textAlignment = alignment  // ✅ 设置对齐方式
        nameLabel.configUI(
            font: UIFont.systemFont(ofSize: 20, weight: .semibold),
            level: level,
            defaultColors: [.white]
        )
    }
}

// 使用时
cell.configure(name: "VIP用户", level: 1, alignment: .left)
```

### 示例 3：根据文字长度动态调整

```swift
func configure(name: String, level: Int) {
    nameLabel.text = name
    
    // 根据文字长度决定对齐方式
    if name.count <= 5 {
        nameLabel.textAlignment = .center  // 短文字居中
    } else {
        nameLabel.textAlignment = .left    // 长文字左对齐
    }
    
    nameLabel.configUI(
        font: UIFont.systemFont(ofSize: 20, weight: .semibold),
        level: level,
        defaultColors: [.white]
    )
}
```

## 总结

**新增属性：** `textAlignment`

**默认值：** `.center`（居中）

**支持的值：** `.left`, `.center`, `.right`, `.justified`, `.natural`

**使用场景：**
- ✅ 短文字不滚动时：控制文字对齐方式
- ✅ 长文字滚动时：不影响滚动效果
- ✅ 可随时动态修改

**推荐配置：**
- 默认使用居中（`.center`）
- 如果需要左对齐，设置 `.left`
- 如果需要右对齐，设置 `.right`

现在你可以完全控制文字的对齐方式了！
